CREATE TABLE tic_sedes_resumen (
    resumen_id INT PRIMARY KEY,
    sede_codigo INT NOT NULL,
    anio INT NOT NULL,
    Total_actividades INT,
    Tiene_internet BOOLEAN,
    Fecha_carga DATETIME,
    CONSTRAINT unique_sede_anio UNIQUE (sede_codigo, anio)
);

SELECT 
    SUBSTR(SEDE_CODIGO, 1, 2) ,
    COUNT(DISTINCT SEDE_CODIGO)
FROM tic_2023
WHERE ACTIVIDAD_CODIGO = '05'
GROUP BY SUBSTR(SEDE_CODIGO, 1, 2)
HAVING COUNT(DISTINCT SEDE_CODIGO) > 500
ORDER BY total_sedes DESC;

-- PUNTO 5
SELECT 
    t22.SEDE_CODIGO AS codigo_sede,
    COUNT(t22.ACTIVIDAD_CODIGO) AS total_act_2022,
    COUNT(t23.ACTIVIDAD_CODIGO) AS total_act_2023,
    (COUNT(t23.ACTIVIDAD_CODIGO) - COUNT(t22.ACTIVIDAD_CODIGO)) AS diferencia,
    CASE 
        WHEN (COUNT(t23.ACTIVIDAD_CODIGO) - COUNT(t22.ACTIVIDAD_CODIGO)) > 0 THEN 'CRECIÓ'
        WHEN (COUNT(t23.ACTIVIDAD_CODIGO) - COUNT(t22.ACTIVIDAD_CODIGO)) < 0 THEN 'DECRECIÓ'
        ELSE 'SIN CAMBIO'
    END AS tendencia
FROM tic_2022 t22
INNER JOIN tic_2023 t23 ON t22.SEDE_CODIGO = t23.SEDE_CODIGO
GROUP BY t22.SEDE_CODIGO
HAVING COUNT(t22.ACTIVIDAD_CODIGO) >= 2
ORDER BY diferencia DESC
LIMIT 30;
